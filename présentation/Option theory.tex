\[  S^{(1)}_t = S^{(1)}_0 exp((\mu_1 - \dfrac{\sigma_1^2}{2})t + \sigma_1 W^{(1)}_t) \]

\[  S^{(2)}_t = S^{(2)}_0 exp((\mu_2 - \dfrac{\sigma_2^2}{2})t + \sigma_2 W^{(2)}_t) \]

\[  S^{(2)}_t = S^{(2)}_0 + \int_0^t \mu_2 S^{(2)}_s ds + \int_0^t \sigma_2 S^{(2)}_s dW^{(2)}_s \]
